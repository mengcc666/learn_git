1) git init

    git clone git@github.com:<username>/<repo_name>.git

2) git add <filename>

3) git commit -m "comment"

4) git diff

5) git log / git reflog

6) git reset --hard <id> 

7) git branch <branch_name>
	
	create branch

8) git checkout <branch_name>

	switch to branch

9) git branch -d <branch_name>
	
	delete branch

10)git switch (-c) <branch_name>

11)git merge <branch_name>

	merge <branch_name> to current branch
